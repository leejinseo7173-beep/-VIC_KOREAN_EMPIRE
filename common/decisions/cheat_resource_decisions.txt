cheat_resource_decisions_open = {
	is_shown = {
		is_player = yes
		NOT = { has_variable = resource_cheat_opened }
	}

	possible = {
		is_player = yes
	}

	when_taken = {
		set_variable = resource_cheat_opened
	}
}

	}
