	name = "dhe_button_purge_corrupt_officials"
	desc = "dhe_button_purge_corrupt_officials_desc"

	visible = {
		c:KOR ?= THIS
		has_journal_entry = je_dhe_sedo_politics_reform
	}

		NOT = { has_variable = dhe_cooldown_purge }
	}

		add_progress = {
			name = dhe_progress_bar_royal_authority

		add_radicals = 10000
	}
}
