dhe_event.3 = {
type = country_event
title = dhe_event.3.t
desc = dhe_event.3.d
icon = "gfx/interface/icons/modifier_icons/custom_modifier_icon.dds"
	event_image = { 
		texture = "gfx/interface/icons/event_pictures/dhe_event_3.dds" 
	}

option = {
		name = dhe_event.3.a
		custom_tooltip = {
			text = dhe_event_3_tt
			tooltip = dhe_event_3_desc_tt
		}
		
		if = {
			limit = { NOT = { has_variable = dhe_reform_score } }
			set_variable = { name = dhe_reform_score value = 0 }
		}
		set_variable = dhe_gwangmu_reform_started

		# ★ 메인 일지 1개
		add_journal_entry = { type = je_dhe_gwangmu_reform_main }
		
		# ★ 분야별 개혁 일지 5개
		add_journal_entry = { type = je_dhe_gwangmu_military }
		add_journal_entry = { type = je_dhe_gwangmu_education }
		add_journal_entry = { type = je_dhe_gwangmu_admin }
		add_journal_entry = { type = je_dhe_gwangmu_society }
		add_journal_entry = { type = je_dhe_gwangmu_infrastructure }

		# [cite_start]★ (수정됨) 4대 권역 산업화 일지 4개 완벽 추가! [cite: 14, 15, 16, 17, 18]
		add_journal_entry = { type = je_korea_capital_industry }
		add_journal_entry = { type = je_korea_south_industry }
		add_journal_entry = { type = je_korea_busan_industry }
		add_journal_entry = { type = je_korea_north_industry }

		# 4대 권역 모디파이어 뿌리기
		every_scope_state = {
			limit = {
				owner = ROOT
				OR = { state_region = s:STATE_SEOUL state_region = s:STATE_HAESEO }
			}
			add_modifier = { name = modifier_capital_developing months = -1 }
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				OR = { state_region = s:STATE_CHNGCHEONG state_region = s:STATE_YANGHO }
			}
			add_modifier = { name = modifier_south_developing months = -1 }
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				state_region = s:STATE_BUSAN
			}
			add_modifier = { name = modifier_busan_developing months = -1 }
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				OR = { state_region = s:STATE_GWANDONG state_region = s:STATE_PYONGYANG state_region = s:STATE_SARIWON }
			}
			add_modifier = { name = modifier_north_developing months = -1 }
		}
	}

}
